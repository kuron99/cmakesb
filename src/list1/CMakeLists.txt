cmake_minimum_required(VERSION 3.10)

set(listval1 aaa;bbb)
set(listval2 ccc;ddd)
foreach(f IN LISTS listval1 listval2)
    message("${f}")
    if(f STREQUAL aaa OR f STREQUAL bbb OR f STREQUAL ccc OR f STREQUAL ddd)
    else()
        message(SEND_ERROR "assert failed")
    endif()
endforeach()

foreach(f IN ITEMS aaa bbb ccc)
    message("${f}")
    if(f STREQUAL aaa OR f STREQUAL bbb OR f STREQUAL ccc)
    else()
        message(SEND_ERROR "assert failed")
    endif()
endforeach()


set(list1 a;b;c)
list(APPEND list1 d)
list(LENGTH list1 list1len)
if (list1len EQUAL 4)
else()
    message(SEND_ERROR "assert failed")
endif()
list(GET list1 1 second)
if (second STREQUAL b)
else()
    message(SEND_ERROR "assert failed")
endif()

list(REMOVE_ITEM list1 b)
list(REMOVE_AT list1 1)
list(GET list1 1 second)
if (second STREQUAL d)
else()
    message(SEND_ERROR "assert failed")
endif()
